let currentPageIdx = 0;
const totalPages = 9;
const pages = document.querySelectorAll('.kobo-page');
const tabs = document.querySelectorAll('.section-tab');
const prevBtn = document.getElementById('prevBtn');
const nextBtn = document.getElementById('nextBtn');
const submitBtn = document.getElementById('submitBtn');
let repeatCount = 0;

function changePage(step) {
  if (step === 1 && !validateCurrentPage()) {
    alert('Mohon isi semua data yang wajib pada halaman ini sebelum melanjutkan.');
    return;
  }
  pages[currentPageIdx].classList.remove('active-page');
  tabs[currentPageIdx].classList.remove('active');
  currentPageIdx += step;
  pages[currentPageIdx].classList.add('active-page');
  tabs[currentPageIdx].classList.add('active');
  prevBtn.disabled = (currentPageIdx === 0);
  if (currentPageIdx === pages.length - 1) {
    nextBtn.style.display = 'none';
    submitBtn.style.display = 'block';
  } else {
    nextBtn.style.display = 'block';
    submitBtn.style.display = 'none';
  }
  updateProgress();
  evaluateAllRelevance();
  window.scrollTo(0, 0);
}

function goToPage(idx) {
  if (idx < 0 || idx >= pages.length) return;
  if (idx > currentPageIdx && !validateCurrentPage()) {
    alert('Mohon isi semua data yang wajib pada halaman ini.');
    return;
  }
  pages[currentPageIdx].classList.remove('active-page');
  tabs[currentPageIdx].classList.remove('active');
  currentPageIdx = idx;
  pages[currentPageIdx].classList.add('active-page');
  tabs[currentPageIdx].classList.add('active');
  prevBtn.disabled = (currentPageIdx === 0);
  if (currentPageIdx === pages.length - 1) {
    nextBtn.style.display = 'none';
    submitBtn.style.display = 'block';
  } else {
    nextBtn.style.display = 'block';
    submitBtn.style.display = 'none';
  }
  updateProgress();
  evaluateAllRelevance();
}

function updateProgress() {
  var pct = ((currentPageIdx + 1) / totalPages) * 100;
  document.getElementById('progressFill').style.width = pct + '%';
  document.getElementById('progressText').textContent = 'Halaman ' + (currentPageIdx + 1) + ' / ' + totalPages;
}

function validateCurrentPage() {
  var activePage = pages[currentPageIdx];
  var fields = activePage.querySelectorAll('input[required], select[required], textarea[required]');
  var isValid = true;
  fields.forEach(function(f) {
    if (f.closest('.question.hidden')) return;
    if (f.type === 'radio') {
      var checked = activePage.querySelector('input[name="' + f.name + '"]:checked');
      if (!checked) isValid = false;
    } else {
      if (!f.value.trim()) isValid = false;
    }
  });
  return isValid;
}

function getFieldValue(name) {
  var radios = document.querySelectorAll('input[type="radio"][name="' + name + '"]');
  if (radios.length > 0) {
    var val = '';
    radios.forEach(function(r) { if (r.checked) val = r.value; });
    return val;
  }
  var el = document.querySelector('select[name="' + name + '"], input[name="' + name + '"], textarea[name="' + name + '"]');
  return el ? el.value : '';
}

function setFieldValue(name, value) {
  if (value === null || value === undefined) value = '';
  var strValue = String(value);
  var radios = document.querySelectorAll('input[type="radio"][name="' + name + '"]');
  if (radios.length > 0) {
    radios.forEach(function(r) {
      r.checked = (r.value === strValue || String(r.value).split('__')[0] === strValue);
      if (r.checked) r.parentElement.classList.add('selected');
      else r.parentElement.classList.remove('selected');
    });
    return;
  }
  var select = document.querySelector('select[name="' + name + '"]');
  if (select) { select.value = strValue; return; }
  var input = document.querySelector('input[name="' + name + '"], textarea[name="' + name + '"]');
  if (input) input.value = strValue;
}

function evaluateAllRelevance() {
  document.querySelectorAll('[data-relevant]').forEach(function(container) {
    var expr = container.getAttribute('data-relevant');
    if (!expr) return;
    try {
      var isRelevant = new Function('getFieldValue', 'return (' + expr + ');')(getFieldValue);
      if (isRelevant) {
        container.classList.remove('hidden');
      } else {
        container.classList.add('hidden');
        container.querySelectorAll('input, select, textarea').forEach(function(el) {
          if (el.type === 'radio') { el.checked = false; el.parentElement.classList.remove('selected'); }
          else el.value = '';
        });
      }
    } catch (e) { console.error(e); }
  });
}

function attachChangeListeners() {
  document.querySelectorAll('input[name], select[name], textarea[name]').forEach(function(el) {
    el.addEventListener('change', evaluateAllRelevance);
    if (el.tagName === 'SELECT' || el.tagName === 'TEXTAREA' || el.type === 'text' || el.type === 'number') {
      el.addEventListener('input', evaluateAllRelevance);
    }
  });
}

document.addEventListener('click', function(e) {
  var opt = e.target.closest('.option-item');
  if (opt) {
    var group = opt.closest('.option-list');
    if (group) {
      group.querySelectorAll('.option-item').forEach(function(o) { o.classList.remove('selected'); });
      opt.classList.add('selected');
      opt.querySelector('input').checked = true;
      evaluateAllRelevance();
    }
  }
});

function handleSubmit(event) {
  event.preventDefault();
  if (!validateCurrentPage()) {
    alert('Mohon isi seluruh field yang diperlukan.');
    return;
  }
  var data = {};
  document.querySelectorAll('input[name], select[name], textarea[name]').forEach(function(el) {
    if (el.type === 'radio') { if (el.checked) data[el.name] = el.value; }
    else if (el.type === 'file') { data[el.name] = el.files[0] ? el.files[0].name : ''; }
    else { data[el.name] = el.value; }
  });
  console.log('Data tersimpan:', data);
  alert('✅ Data berhasil disimpan! (Lihat console untuk detail)');
}

function removeRepeatItem(idx) {
  var item = document.getElementById('repeat_' + idx);
  if (item) item.remove();
}

function evaluateRepeatRelevance() {
  document.querySelectorAll('.repeat-item .question[data-relevant]').forEach(function(group) {
    var expr = group.getAttribute('data-relevant');
    if (!expr) return;
    var idx = group.getAttribute('data-repeat-idx');
    try {
      var getVal = function(name) {
        var el = document.querySelector('[name="' + name + '_' + idx + '"]');
        if (el) return el.value;
        return getFieldValue(name);
      };
      var result = new Function('getFieldValue', 'return (' + expr + ');')(getVal);
      if (result) {
        group.classList.remove('hidden');
        group.querySelectorAll('input, select, textarea').forEach(function(el) { el.disabled = false; });
      } else {
        group.classList.add('hidden');
        group.querySelectorAll('input, select, textarea').forEach(function(el) { el.disabled = true; });
      }
    } catch(e) { console.warn('Repeat relevance error:', e.message); }
  });
}

function prefillData(data) {
  Object.entries(data).forEach(function(entry) {
    setFieldValue(entry[0], entry[1]);
  });
  evaluateAllRelevance();
}
